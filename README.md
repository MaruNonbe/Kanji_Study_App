function doGet(e) {
  const ss = SpreadsheetApp.getActive();
  const gradeFilter = e && e.parameter && e.parameter.grade
    ? String(e.parameter.grade)
    : '';

  let allItems = [];

  SHEET_NAMES.forEach(name => {
    const sh = ss.getSheetByName(name);
    if (!sh) return;
    const values = sh.getDataRange().getValues();
    if (values.length < 2) return;

    const header = values[0];
    const body   = values.slice(1);

    body.forEach(row => {
      if (!row[0]) return; // kanji ãŒç©ºãªã‚‰ã‚¹ã‚­ãƒƒãƒ—

      allItems.push(obj);
    });
  });

  if (gradeFilter) {
    allItems = allItems.filter(r => String(r.grade) === gradeFilter);
  }

  return ContentService
    .createTextOutput(JSON.stringify({ items: allItems }))
    .setMimeType(ContentService.MimeType.JSON);
